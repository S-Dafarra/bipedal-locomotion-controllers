164 template <typename T> void mergeSigVector(yarp::sig::Vector& vector, const T& t)
165 {
166  if constexpr (std::is_arithmetic<T>::value)
167  vector.push_back(t);
168 
169  else if constexpr (!std::is_arithmetic<T>::value)
170  {
171  using elementType = typename std::pointer_traits<decltype(t.data())>::element_type;
172  static_assert(std::is_convertible<elementType, double>::value,
173  "[BipedalLocomotionControllers::YarpUtilities::mergeSigVector] The element "
174  "contained in the vector cannot be converted in a double");
175 
176  if constexpr (is_iterable<T>::value)
177  {
178  for (auto it = t.begin(); it != t.end(); it++)
179  vector.push_back(*it);
180 
181  } else if constexpr (has_square_bracket_operator<T>::value)
182  {
183  for (int i = 0; i < t.size(); i++)
184  vector.push_back(t[i]);
185 
186  } else
187  static_assert(dependent_false<T>::value,
188  "[BipedalLocomotionControllers::YarpUtilities::mergeSigVector] The "
189  "Vector type does not have square bracket operator nor begin()/end() "
190  "methods");
191  return;
192  }
193 
194  else
195  {
196  static_assert(dependent_false<T>::value,
197  "[BipedalLocomotionControllers::YarpUtilities::mergeSigVector] The type of "
198  "the input element cannot be handled by the function");
199  }

201  return;
